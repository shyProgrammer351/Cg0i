문제

A양과 B군이 가위바위보를 선택하면 승패를 출력하는 문제입니다 실행순서를 클릭하세요 
A양, 가위, 바위, 보 중 하나를 선택하세요: 바위
바위를 냈습니다.
B군, 가위, 바위, 보 중 하나를 선택하세요: 가위
가위를 냈습니다.
A양이 이겼습니다.

프로그램 코드

	#include <stdio.h>

	#include <string.h>


1	int main() {

		printf( "A양, 가위, 바위, 보 중 하나를 선택하세요: " );

		char playerA[10] = "";

		scanf( "%s", playerA );

2		printf( "%s를 냈습니다.\n", playerA );


		printf( "B군, 가위, 바위, 보 중 하나를 선택하세요: " );

		char playerB[10] = "";

		scanf( "%s", playerB );

3		printf( "%s를 냈습니다.\n", playerB );


4		if ( !strcmp( playerA, playerB ) ) {

5			printf( "\nA양과 B군이 비겼습니다.\n" );

		}

6		else if ( !strcmp( playerA, "가위" ) && !strcmp( playerB, "보" ) )	 {

7			printf( "\nA양이 이겼습니다.\n" );

		}

8		else if ( !strcmp( playerA, "바위" ) && !strcmp( playerB, "가위" ) ) {

9			printf( "\nA양이 이겼습니다.\n" );

		}

10		else if ( !strcmp( playerA, "보" ) && !strcmp( playerB, "바위" ) ) {

11			printf( "\nA양이 이겼습니다.\n" );

		}

12		else {

13			printf( "\nB군이 이겼습니다.\n" );

		}


14		return 0;

	}
