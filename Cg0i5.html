알고리즘

프로그램 시작 
   단어들 입력 
   단어 정렬 
      마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 
         영역의 첫번째 칸부터 인접한 두 값을 비교하면서 
         앞칸 단어가 뒤칸 단어보다 알파벳순으로 뒤면, 두 단어를 맞바꾸기 
   정렬된 단어 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter10/WordSort.c

	#include <stdio.h>

	#include <string.h>

	 

	// 프로그램 시작 

1	int main() {

2		char words[3][32] = { "", "", "" };

		 

		// 단어들 입력  

3		for ( int index = 0; index < 3; index++ ) {

4			printf( "단어를 입력하세요: " );

5			scanf( "%s", words[index] );

		}

 

		// 단어 정렬 

		// 마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 

6		for ( int last = 2; 0 < last; last-- ) {

			// 영역의 첫번째 칸부터 인접한 두 값을 비교하면서  

7			for ( int index = 0; index < last; index++ ) {

				// 앞칸 단어가 뒤칸 단어보다 알파벳순으로 뒤면, 두 단어를 맞바꾸기 

8				if ( strcmp( words[index], words[index+1] ) > 0 ) {

					char temp[32] = "";

9					strcpy( temp, words[index] );

10					strcpy( words[index], words[index+1] );

11					strcpy( words[index+1], temp );

				}

			}

		}

	 

		// 정렬된 단어 출력 

12		for ( int index = 0; index < 3; index++ )

13			printf( "%s ", words[index] );

	 

		// 프로그램 종료 

14		return 0;

	}
